```kotlin
fun climbStairsForIter(n: Int): Int =
    if (n <= 2) n
    else {
        var acc1 = 1
        var acc2 = 2
        (3 .. n).forEach { _ ->
            val temp = acc2
            acc2 += acc1
            acc1 = temp
        }
        acc2
    }
```

```kotlin
tailrec fun climbStairs(n: Int, acc1: Int = 1, acc2: Int = 2): Int = when (n) {
```
